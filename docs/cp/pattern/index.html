Patterns#

1. Square Star#

int i=0,j=0;
while (i<n){
while(j<n){
System.out.print("* ");
j++;
}
j=0;
i++;
System.out.println("");
}

2. continues number matrix#

int k =1;
int i=0;
while(i<n){
int j =0;
while(j<n){
System.out.print(k+" ");
k++;
j++;
}
System.out.println();
i++;
}

3. Star triangle#

int row=1;
while(row<=n){
int col =0;
while(col<row){
System.out.print("* ");
col++;
}
row++;
System.out.println();
}

9.#

int row=1;
int col = 1;
while(row<=n){
//print space
while(col<=n-row){
System.out.print(" ");
col++;
}
col=1;
//print 1st triangle
while(col<=row){
System.out.print(col);
col++;
}
//print 2nd triangle
int start = row-1;
while(start>0){
System.out.print(start);
start--;
}
col=1;
row++;
System.out.println();
}
